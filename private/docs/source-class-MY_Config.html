<?php  if ( ! defined('BASEPATH')) exit('No direct script access allowed');

/**
 * Class MY_Config
 * @package BIBSN\Core
 */
class MY_Config extends CI_Config {

    /**
     * @param string $uri
     * @return string
     */
    function site_url($uri = '')
    {    
        if (is_array($uri))
        {
            $uri = implode('/', $uri);
        }

        if (function_exists('get_instance'))        
        {
            $CI =& get_instance();
            $uri = $CI->lang->localized($uri);            
        }

        return parent::site_url($uri);
    }

}
